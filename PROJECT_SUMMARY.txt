DATABASE SCHEMA - WHAT'S NEW
================================================================================

USERS TABLE:
+ is_active TINYINT(1) DEFAULT 1
  Purpose: Soft delete support

PRODUCTS TABLE:
+ payment_methods JSON DEFAULT '["cod","upi"]'
  Purpose: Set payment methods per product

ORDERS TABLE:
+ delivery_otp_generated_at DATETIME
+ otp_verified TINYINT(1) DEFAULT 0
+ delivered_at DATETIME
+ delivery_person_name VARCHAR(120)
+ delivery_person_phone VARCHAR(20)
  Purpose: Enhanced delivery tracking

OFFER CODES TABLE:
+ discount_type ENUM('fixed', 'percentage') DEFAULT 'fixed'
+ discount_value DECIMAL(10,2) DEFAULT 0
  Purpose: Support fixed and percentage discounts

INDEXES ADDED:
+ idx_orders_status_user ON orders(status, user_id)
+ idx_offer_codes_active ON offer_codes(is_active, expires_at)
+ idx_products_active ON products(is_active)
+ idx_cart_user ON cart_items(user_id)
+ idx_users_email ON users(email)
  Purpose: Query performance optimization
